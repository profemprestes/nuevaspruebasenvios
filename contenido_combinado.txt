        document.addEventListener('DOMContentLoaded', () => {
            // Objeto de utilidades para funciones comunes
            const utils = {
                // Funci√≥n para activar/desactivar clases
                toggleClass: (element, className, force) => {
                    if (element) {
                        if (force !== undefined) {
                            element.classList[force ? 'add' : 'remove'](className);
                        } else {
                            element.classList.toggle(className);
                        }
                    }
                },
                
                // Funci√≥n para agregar eventos con verificaci√≥n de existencia
                addSafeEventListener: (element, event, callback) => {
                    if (element) {
                        element.addEventListener(event, callback);
                    }
                },
                
                // Funci√≥n para seleccionar elementos con seguridad
                select: (selector, parent = document) => parent.querySelector(selector),
                selectAll: (selector, parent = document) => parent.querySelectorAll(selector)
            };

            // Navegaci√≥n m√≥vil
            const initMobileNav = () => {
                const mobileNavToggle = utils.select('.mobile-nav-toggle');
                const navMenu = utils.select('.nav-menu');
                const dropdowns = utils.selectAll('.dropdown');
                
                // Toggle del men√∫ m√≥vil
                utils.addSafeEventListener(mobileNavToggle, 'click', () => {
                    utils.toggleClass(navMenu, 'active');
                    utils.toggleClass(document.body, 'nav-open');
                });
                
                // Manejo de dropdowns
                dropdowns.forEach(dropdown => {
                    const link = utils.select('a', dropdown);
                    
                    utils.addSafeEventListener(link, 'click', (e) => {
                        if (window.innerWidth <= 768) {
                            e.preventDefault();
                            utils.toggleClass(dropdown, 'active');
                            
                            // Cerrar otros dropdowns
                            dropdowns.forEach(other => {
                                if (other !== dropdown) {
                                    utils.toggleClass(other, 'active', false);
                                }
                            });
                        }
                    });
                });
                
                // Cerrar men√∫ al hacer clic fuera
                document.addEventListener('click', (e) => {
                    if (!e.target.closest('.nav-menu') && !e.target.closest('.mobile-nav-toggle')) {
                        utils.toggleClass(navMenu, 'active', false);
                        utils.toggleClass(document.body, 'nav-open', false);
                    }
                });
                
                // Ajustar en cambio de tama√±o
                window.addEventListener('resize', () => {
                    if (window.innerWidth > 768) {
                        dropdowns.forEach(dropdown => utils.toggleClass(dropdown, 'active', false));
                    }
                });
            };

            // Funcionalidad de FAQ
            const initFAQ = () => {
                const pills = utils.selectAll('.nav-pill');
                const sections = utils.selectAll('.faq-section');
                const accordionTabs = utils.selectAll('.ss-accordion-tab');
                const searchInput = document.getElementById('searchFaq');
                
                // Navegaci√≥n por pesta√±as
                pills.forEach(pill => {
                    utils.addSafeEventListener(pill, 'click', () => {
                        // Desactivar todas las pesta√±as y activar la actual
                        pills.forEach(p => utils.toggleClass(p, 'active', false));
                        utils.toggleClass(pill, 'active', true);
                        
                        // Mostrar/ocultar secciones seg√∫n categor√≠a
                        const category = pill.dataset.category;
                        sections.forEach(section => {
                            section.style.display = (category === 'todos' || section.id === category) ? 'block' : 'none';
                        });
                    });
                });
                
                // Funcionalidad de acorde√≥n
                accordionTabs.forEach(tab => {
                    const btn = utils.select('.ss-accordion-btn', tab);
                    
                    utils.addSafeEventListener(btn, 'click', () => {
                        const wasActive = tab.classList.contains('active');
                        
                        // Cerrar todos los tabs
                        accordionTabs.forEach(t => utils.toggleClass(t, 'active', false));
                        
                        // Abrir el actual si no estaba abierto
                        if (!wasActive) {
                            utils.toggleClass(tab, 'active', true);
                        }
                    });
                });
                
                // B√∫squeda en FAQ
                utils.addSafeEventListener(searchInput, 'input', (e) => {
                    const searchTerm = e.target.value.toLowerCase().trim();
                    
                    accordionTabs.forEach(tab => {
                        const question = utils.select('.ss-accordion-btn', tab)?.textContent.toLowerCase() || '';
                        const answer = utils.select('.content', tab)?.textContent.toLowerCase() || '';
                        const matches = question.includes(searchTerm) || answer.includes(searchTerm);
                        
                        // Mostrar/ocultar seg√∫n coincidencia
                        tab.style.display = (searchTerm === '' || matches) ? 'block' : 'none';
                        
                        // Expandir si hay coincidencia
                        utils.toggleClass(tab, 'active', matches && searchTerm !== '');
                    });
                });
            };

            // Animaciones al hacer scroll
            const initScrollAnimations = () => {
                const observerOptions = {
                    threshold: 0.1,
                    rootMargin: '0px 0px -50px 0px'
                };
                
                const observer = new IntersectionObserver((entries) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting) {
                            utils.toggleClass(entry.target, 'animate', true);
                        }
                    });
                }, observerOptions);
                
                // Observar elementos para animaci√≥n
                utils.selectAll('.vision-card, .service-card, .feature-card').forEach(card => {
                    observer.observe(card);
                });
            };

            // Bot√≥n volver arriba
            const initBackToTop = () => {
                const backToTop = document.getElementById('backToTop');
                
                if (backToTop) {
                    // Mostrar/ocultar bot√≥n seg√∫n posici√≥n de scroll
                    window.addEventListener('scroll', () => {
                        const scrollPosition = document.documentElement.scrollTop || document.body.scrollTop;
                        backToTop.classList.toggle('visible', scrollPosition > 20);
                    });
                    
                    // Acci√≥n de volver arriba
                    utils.addSafeEventListener(backToTop, 'click', () => {
                        window.scrollTo({
                            top: 0,
                            behavior: 'smooth'
                        });
                    });
                }
            };

            // Scroll suave para enlaces internos
            const initSmoothScroll = () => {
                utils.selectAll('a[href^="#"]:not([href="#"])').forEach(anchor => {
                    utils.addSafeEventListener(anchor, 'click', function(e) {
                        e.preventDefault();
                        const targetId = this.getAttribute('href');
                        const targetElement = document.querySelector(targetId);
                        
                        if (targetElement) {
                            targetElement.scrollIntoView({
                                behavior: 'smooth',
                                block: 'start'
                            });
                        }
                    });
                });
            };

            // Inicializaci√≥n de carrusel Bootstrap
            const initCarousel = () => {
                const carousel = utils.select('.carousel');
                
                if (carousel && typeof bootstrap !== 'undefined') {
                    new bootstrap.Carousel(carousel, {
                        interval: 4000,
                        pause: 'hover'
                    });
                }
            };

            // Inicializar todas las funcionalidades
            const init = () => {
                initMobileNav();
                initFAQ();
                initScrollAnimations();
                initBackToTop();
                initSmoothScroll();
                initCarousel();
            };

            // Ejecutar inicializaci√≥n
            init();
        });
